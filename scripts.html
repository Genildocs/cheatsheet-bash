📊 Scripts de Monitoramento
Monitor de Recursos do Sistema
Monitora CPU, memória e disco em tempo real
#!/bin/bash # system-monitor.sh - Monitor de recursos do sistema while true; do clear echo "=== MONITOR DE RECURSOS ===" echo "Data/Hora: $(date '+%d/%m/%Y %H:%M:%S')" echo "" # Uso de CPU CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1) echo "🔥 CPU: ${CPU_USAGE}%" # Uso de Memória MEM_INFO=$(free -h | grep Mem) MEM_TOTAL=$(echo $MEM_INFO | awk '{print $2}') MEM_USED=$(echo $MEM_INFO | awk '{print $3}') MEM_PERCENT=$(free | grep Mem | awk '{printf "%.1f", $3/$2 * 100.0}') echo "💾 Memória: $MEM_USED / $MEM_TOTAL (${MEM_PERCENT}%)" # Uso de Disco DISK_USAGE=$(df -h / | tail -1 | awk '{print $5}') echo "💿 Disco: ${DISK_USAGE} usado" # Processos em execução PROCESS_COUNT=$(ps aux | wc -l) echo "⚙️ Processos: $PROCESS_COUNT" # Load Average LOAD_AVG=$(uptime | awk -F'load average:' '{print $2}') echo "📊 Load Average:${LOAD_AVG}" echo "" echo "Pressione Ctrl+C para sair..." sleep 5 done

#!/bin/bash # service-checker.sh - Verifica status de serviços SERVICES=("nginx" "mysql" "postgresql" "redis-server" "docker") echo "=== VERIFICAÇÃO DE SERVIÇOS ===" echo "Data: $(date '+%d/%m/%Y %H:%M:%S')" echo "" for service in "${SERVICES[@]}"; do if systemctl is-active --quiet $service; then STATUS="✅ ATIVO" COLOR="\033[32m" elif systemctl list-unit-files | grep -q "^${service}.service"; then STATUS="❌ INATIVO" COLOR="\033[31m" else STATUS="⚠️ NÃO INSTALADO" COLOR="\033[33m" fi echo -e "${COLOR}${service}: ${STATUS}\033[0m" done # Verifica portas específicas echo "" echo "=== VERIFICAÇÃO DE PORTAS ===" PORTS=(80 443 3306 5432 6379 27017) for port in "${PORTS[@]}"; do if netstat -tuln | grep -q ":$port "; then STATUS="✅ ABERTA" COLOR="\033[32m" else STATUS="❌ FECHADA" COLOR="\033[31m" fi echo -e "${COLOR}Porta $port: ${STATUS}\033[0m" done

💾 Scripts de Backup
Backup Automático com Rotação
Cria backups com data e mantém histórico
#!/bin/bash # backup-with-rotation.sh - Backup com rotação automática SOURCE_DIR="/var/www/html" BACKUP_DIR="/backups" MAX_BACKUPS=7 TIMESTAMP=$(date '+%Y%m%d_%H%M%S') BACKUP_NAME="backup_${TIMESTAMP}.tar.gz" # Cria diretório de backup se não existir mkdir -p $BACKUP_DIR # Cria backup compactado echo "Criando backup: $BACKUP_NAME" tar -czf "${BACKUP_DIR}/${BACKUP_NAME}" -C "$(dirname $SOURCE_DIR)" "$(basename $SOURCE_DIR)" # Verifica se o backup foi criado com sucesso if [ $? -eq 0 ]; then echo "Backup criado com sucesso!" # Remove backups antigos cd $BACKUP_DIR BACKUP_COUNT=$(ls backup_*.tar.gz 2>/dev/null | wc -l) if [ $BACKUP_COUNT -gt $MAX_BACKUPS ]; then echo "Removendo backups antigos..." ls -t backup_*.tar.gz | tail -n +$((MAX_BACKUPS + 1)) | xargs rm -f echo "Rotação de backup concluída." fi # Lista backups disponíveis echo "" echo "Backups disponíveis:" ls -lh backup_*.tar.gz else echo "ERRO: Falha ao criar backup!" exit 1 fi
Backup de Banco de Dados MySQL
Backup automatizado de banco de dados MySQL
#!/bin/bash # mysql-backup.sh - Backup de banco de dados MySQL DB_HOST="localhost" DB_USER="backup_user" DB_PASS="senha_segura" DB_NAME="meu_banco" BACKUP_DIR="/backups/mysql" TIMESTAMP=$(date '+%Y%m%d_%H%M%S') BACKUP_FILE="${DB_NAME}_${TIMESTAMP}.sql.gz" # Configurações RETENTION_DAYS=30 # Cria diretório de backup mkdir -p $BACKUP_DIR # Realiza backup echo "Iniciando backup do banco: $DB_NAME" mysqldump -h $DB_HOST -u $DB_USER -p$DB_PASS \ --single-transaction \ --routines \ --triggers \ $DB_NAME | gzip > "${BACKUP_DIR}/${BACKUP_FILE}" # Verifica sucesso if [ ${PIPESTATUS[0]} -eq 0 ]; then echo "Backup concluído: $BACKUP_FILE" # Remove backups antigos find $BACKUP_DIR -name "${DB_NAME}_*.sql.gz" -mtime +$RETENTION_DAYS -delete echo "Backups mais antigos que $RETENTIONdias dias removidos." # Mostra espaço utilizado echo "Espaço utilizado: $(du -sh $BACKUP_DIR)" else echo "ERRO: Falha no backup do banco!" exit 1 fi

Analisador de Logs em Tempo Real
Monitora logs e extrai informações relevantes
#!/bin/bash # log-analyzer.sh - Analisador de logs em tempo real LOG_FILE=${1:-"/var/log/nginx/access.log"} PATTERNS_FILE="patterns.conf" ALERT_THRESHOLD=10 # Padrões para buscar declare -A PATTERNS=( ["ERROR"]="error|Error|ERROR" ["404"]=" 404 " ["500"]=" 500 " ["SUSPICIOUS"]="union|select|drop|insert|update|delete" ) # Cria arquivo de padrões se não existir if [ ! -f "$PATTERNS_FILE" ]; then for pattern in "${!PATTERNS[@]}"; do echo "$pattern=${PATTERNS[$pattern]}" >> $PATTERNS_FILE done fi # Carrega padrões declare -A COUNTS for pattern in "${!PATTERNS[@]}"; do COUNTS[$pattern]=0 done echo "Analisando log: $LOG_FILE" echo "Pressione Ctrl+C para parar" echo "" # Monitora o arquivo em tempo real tail -f $LOG_FILE | while read line; do timestamp=$(date '+%Y-%m-%d %H:%M:%S') for pattern in "${!PATTERNS[@]}"; do if echo "$line" | grep -qE "${PATTERNS[$pattern]}"; then COUNTS[$pattern]=$((${COUNTS[$pattern]} + 1)) # Alerta se ultrapassar threshold if [ ${COUNTS[$pattern]} -gt $ALERT_THRESHOLD ]; then echo "🚨 ALERTA: Muitas ocorrências de $pattern!" echo "Última linha: $line" echo "" fi echo "[$timestamp] $pattern: ${COUNTS[$pattern]} ocorrências" fi done done
Deploy Automático com Rollback
Script de deploy com capacidade de rollback
#!/bin/bash # deploy-with-rollback.sh - Deploy automático com rollback APP_DIR="/var/www/myapp" BACKUP_DIR="/backups/deploy" BRANCH=${1:-"main"} SERVICE_NAME="myapp" # Função de rollback rollback() { echo "🔄 Iniciando rollback..." if [ -d "$BACKUP_DIR/current" ]; then # Para o serviço systemctl stop $SERVICE_NAME # Restaura backup rm -rf $APP_DIR cp -r $BACKUP_DIR/current $APP_DIR # Reinicia o serviço systemctl start $SERVICE_NAME echo "✅ Rollback concluído com sucesso!" else echo "❌ Nenhum backup encontrado para rollback!" exit 1 fi } # Verifica se é rollback if [ "$1" = "rollback" ]; then rollback exit 0 fi # Cria backup do estado atual echo "💾 Criando backup do estado atual..." mkdir -p $BACKUP_DIR rm -rf $BACKUP_DIR/previous mv $BACKUP_DIR/current $BACKUP_DIR/previous 2>/dev/null cp -r $APP_DIR $BACKUP_DIR/current # Atualiza código echo "📥 Baixando atualizações da branch $BRANCH..." cd $APP_DIR git fetch origin git checkout $BRANCH git pull origin $BRANCH # Instala dependências echo "📦 Instalando dependências..." npm ci --production # Executa testes echo "🧪 Executando testes..." npm test if [ $? -ne 0 ]; then echo "❌ Testes falharam! Iniciando rollback..." rollback exit 1 fi # Build do projeto echo "🔨 Build do projeto..." npm run build if [ $? -ne 0 ]; then echo "❌ Build falhou! Iniciando rollback..." rollback exit 1 fi # Reinicia serviço echo "🔄 Reiniciando serviço..." systemctl restart $SERVICE_NAME # Verifica se o serviço está rodando sleep 5 if systemctl is-active --quiet $SERVICE_NAME; then echo "✅ Deploy concluído com sucesso!" # Limpa backup anterior rm -rf $BACKUP_DIR/previous # Registra log echo "$(date '+%Y-%m-%d %H:%M:%S') - Deploy bem-sucedido da branch $BRANCH" >> $BACKUP_DIR/deploy.log else echo "❌ Serviço não iniciou! Iniciando rollback..." rollback exit 1 fi

💡 DICA

Para scripts de deploy em produção, sempre implemente verificação de saúde (health check) antes de considerar o deploy como concluído.

Use quando: Precisa de automação avançada com tratamento de erros e recuperação
